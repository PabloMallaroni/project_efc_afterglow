def load_nifti_files(file_paths):
    images = [nib.load(file_path) for file_path in file_paths]
    return images

def average_nifti_images(images):
  
    data = np.mean([img.get_fdata() for img in images], axis=0)
    affine = images[0].affine
    averaged_img = new_img_like(images[0], data, affine)
    return averaged_img


def find_peak_mni_coords(image, per_hemisphere=False):

    data = image.get_fdata()
    filtered_data = maximum_filter(data, size=3)
    
    if per_hemisphere:
        # Separate the data into left and right hemispheres based on MNI coordinates
        coords = np.array(np.nonzero(data)).T
        mni_coords = nib.affines.apply_affine(image.affine, coords)
        
        left_coords = coords[mni_coords[:, 0] < 0]
        right_coords = coords[mni_coords[:, 0] >= 0]
        
        # Find peak in left hemisphere
        left_peak_values = data[left_coords[:, 0], left_coords[:, 1], left_coords[:, 2]]
        left_peak_index = np.argmax(left_peak_values)
        left_peak_coord = left_coords[left_peak_index]
        left_peak_mni_coord = nib.affines.apply_affine(image.affine, left_peak_coord)
        
        # Find peak in right hemisphere
        right_peak_values = data[right_coords[:, 0], right_coords[:, 1], right_coords[:, 2]]
        right_peak_index = np.argmax(right_peak_values)
        right_peak_coord = right_coords[right_peak_index]
        right_peak_mni_coord = nib.affines.apply_affine(image.affine, right_peak_coord)
        
        return left_peak_mni_coord, right_peak_mni_coord
    
    else:
        peak_coords = np.where(data == filtered_data)
        peak_values = data[peak_coords]
        
        # Convert voxel coordinates to MNI coordinates
        mni_coords = [nib.affines.apply_affine(image.affine, voxel) for voxel in zip(*peak_coords)]
        
        # Extract the singular peak coordinate with the highest value
        peak_value = max(peak_values)
        peak_index = peak_values.tolist().index(peak_value)
        peak_mni_coord = mni_coords[peak_index]
        
        return peak_mni_coord

# Paths to your NeuroSynth NIfTI files
path_main = '/Users/administrator/Documents/MATLAB/project_cb_tom/neurosynth_maps_dmn'

regions = {
    'vmpfc': glob.glob(os.path.join(path_main, 'vmpfc', '*.nii.gz')),
    'dmpfc': glob.glob(os.path.join(path_main, 'dmpfc', '*.nii.gz')),
    'precuneus': glob.glob(os.path.join(path_main, 'precuneus', '*.nii.gz')),
    'tpj': glob.glob(os.path.join(path_main, 'tpj', '*.nii.gz'))
}

# Initialize an empty list to store the results
results = []

# Process each region
for region_name, file_paths in regions.items():
    # Load the NIfTI files
    images = load_nifti_files(file_paths)
    
    # Average the NIfTI images
    averaged_image = average_nifti_images(images)
    
    # Find the peak coordinates
    if region_name == 'tpj':
        left_peak_mni_coord, right_peak_mni_coord = find_peak_mni_coords(averaged_image, per_hemisphere=True)
        results.append({'Region': f'{region_name}_left', 'X': left_peak_mni_coord[0], 'Y': left_peak_mni_coord[1], 'Z': left_peak_mni_coord[2]})
        results.append({'Region': f'{region_name}_right', 'X': right_peak_mni_coord[0], 'Y': right_peak_mni_coord[1], 'Z': right_peak_mni_coord[2]})
    else:
        peak_mni_coord = find_peak_mni_coords(averaged_image, per_hemisphere=False)
        results.append({'Region': region_name, 'X': peak_mni_coord[0], 'Y': peak_mni_coord[1], 'Z': peak_mni_coord[2]})

# Create a DataFrame to display the results
results_df = pd.DataFrame(results)
# Save the dataframe to a CSV file
results_df.to_csv(os.path.join(path_main,'neurosynth_coord.csv'), index=False)
print(results_df)
